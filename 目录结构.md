在项目或auto.props中套用以下配置:
```xml:auto.props 或 *.vcxproj
<?xml version="1.0" encoding="utf-8"?>
<Project ToolsVersion="4.0" xmlns="http://schemas.microsoft.com/developer/msbuild/2003">
  <!-- 静态库项目配置 -->
  <PropertyGroup Label="UserMacros" Condition="'$(ConfigurationType)' == 'StaticLibrary'">
    <OutDir>$(SolutionDir)lib\$(Platform)\$(Configuration)\</OutDir>
    <IntDir>$(SolutionDir)temp\$(Platform)\$(Configuration)\$(ProjectName)\</IntDir>
  </PropertyGroup>
  <!-- 应用程序项目配置 -->
  <PropertyGroup Label="UserMacros" Condition="'$(ConfigurationType)' == 'Application'">
    <OutDir>$(SolutionDir)bin\$(Platform)\$(Configuration)\</OutDir>
    <IntDir>$(SolutionDir)temp\$(Platform)\$(Configuration)\$(ProjectName)\</IntDir>
  </PropertyGroup>
  <!-- 编译器设置 -->
  <PropertyGroup>
    <LanguageStandard>stdcpp17</LanguageStandard>
    <LanguageStandard_C>stdc17</LanguageStandard_C>
    <CharacterSet>Unicode</CharacterSet>
    <PlatformToolset>v142</PlatformToolset>
    <WindowsTargetPlatformVersion>10.0</WindowsTargetPlatformVersion>
  </PropertyGroup>
  <ItemDefinitionGroup>
    <ClCompile>
      <LanguageStandard>stdcpp17</LanguageStandard>
      <LanguageStandard_C>stdc17</LanguageStandard_C>
      <AdditionalOptions>/source-charset:utf-8 /execution-charset:utf-8 %(AdditionalOptions)</AdditionalOptions>
      <Optimization Condition="'$(Configuration)'=='Release'">MaxSpeed</Optimization>
      <FavorSizeOrSpeed Condition="'$(Configuration)'=='Release'">Size</FavorSizeOrSpeed>
      <WholeProgramOptimization>false</WholeProgramOptimization>
      <WarningLevel>Level4</WarningLevel>
    </ClCompile>
  </ItemDefinitionGroup>
  <!-- 包含路径设置 -->
  <PropertyGroup>
    <IncludePath>$(SolutionDir)lib;$(IncludePath)</IncludePath>
    <LibraryPath>$(SolutionDir)lib\$(Platform)\$(Configuration);$(LibraryPath)</LibraryPath>
  </PropertyGroup>
  <!-- Release 配置 -->
  <PropertyGroup Condition="'$(Configuration)'=='Release'">
    <InlineFunctionExpansion>OnlyExplicitInline</InlineFunctionExpansion>
    <IntrinsicFunctions>true</IntrinsicFunctions>
    <DebugInformation>None</DebugInformation>
    <DebugInformationFormat>None</DebugInformationFormat>
  </PropertyGroup>
  <!-- Debug 配置 -->
  <PropertyGroup Condition="'$(Configuration)'=='Debug'">
    <DebugInformation>true</DebugInformation>
    <DebugInformationFormat>EditAndContinue</DebugInformationFormat>
  </PropertyGroup>
</Project>
```


```tree:$foldername 为当前项目文件夹的名称

(IDE工作根目录)
└── 解决方案工作区/                    # 主解决方案目录 (名称由变量 $foldername 表示)
    │
    ├── auto.props                   # 公共属性表，统一管理项目编译设置
    ├── *.sln                         # Visual Studio 解决方案文件
    ├── 项目工作区/                   # 主要项目容器
    │   │
    │   ├── .project/                 # 项目相关文档目录
    │   │   ├── docs/                 # 常规文档
    │   │   ├── *历史参考文档/        # 存档的、有参考价值的历史资料
    │   │   └── 避坑指南.md           # 记录常见问题和解决方案的实用文档
    │   │
    │   ├── script/                     # 脚本、批处理文件目录，严禁污染其他目录
    │   ├── test/                       # 临时测试文件、输出专用目录，严禁污染其他目录
    │   ├── log/                        # 编译日志存放目录，严禁污染其他目录
    │   │
    │   ├── $p1/                      # 项目1目录 (只允许写入项目1的项目文件和源码，严禁污染)
    │   │   ├── p1.vcxproj            # 项目1的 Visual Studio 项目文件
    │   │   └── p1_mvp.cpp            # 项目1的“最小可行产品”核心代码文件
    │   │
    │   ├── $lib1/                    # 第三方库1的集成项目目录 (例如:集成spdlog)
    │   │   └── lib1.vcxproj          # 用于编译或包含第三方库1的项目文件
    │   │
    │   ├── temp/                     # 解决方案级公共临时文件目录 (如编译中间文件.obj)
    │   │   └── *.obj
    │   │
    │   ├── bin/                      # 最终二进制输出目录 (由 auto.props 配置)
    │   │   └── *.exe, *.dll          # 存放编译生成的文件，$(SolutionDir)bin\$(Platform)\$(Configuration)
    │   │
    │   └── lib/                         # 静态库专用目录 $(SolutionDir)lib\$(Platform)\$(Configuration)
    │           └── ...                   # 根据上述静态库配置，由编译器自动管理
    │
    └── (可能还有其他项目或目录)
```
